    <script>
  function toggleSection(sectionId) {
    // Get the currently opened section, if any
    const openedSection = document.querySelector('.checkbox-container:not([style*="display: none"])');
    const clickedSection = document.getElementById(`section-${sectionId}`);
    const clickedHeading = document.getElementById(`headings-${sectionId}`);
    const allHeadings = document.querySelectorAll('.headings');

    // If there is an opened section and it's not the clicked one, close it
    if (openedSection && openedSection !== clickedSection) {
      openedSection.style.display = 'none';
      allHeadings.forEach(headings => headings.classList.remove('active'));
    }

    // Toggle the clicked section
    if (clickedSection.style.display === 'none' || clickedSection.style.display === '') {
      clickedSection.style.display = 'block';
      clickedHeading.classList.add('active');
    } else {
      clickedSection.style.display = 'none';
      clickedHeading.classList.remove('active');
    }
  }
</script>
